💻 実装例

1. PAC CLI 初期化時の設定

pac code init --displayName "My Code App" -l "./assets/logo.svg"

2. React コンポーネントでの利用

import React from 'react'; export const AppHeader: React.FC = () => { return ( <header className="flex items-center gap-3 p-4 bg-blue-600 text-white"> <img src="/assets/logo.svg" alt="Code App Logo" className="w-8 h-8" /> <h1 className="text-xl font-semibold">My Code App</h1> </header> ); };

3. HTML での直接利用

<head> <link rel="icon" type="image/svg+xml" href="/assets/favicon.svg"> <title>My Code App</title> </head>

4. TailwindCSS テーマ統合

// tailwind.config.js module.exports = { theme: { extend: { colors: { 'code-apps': { primary: '#0078D4', // ロゴメインカラー secondary: '#005A9E', // ロゴボーダー accent: '#FFD700', // ライトニング } } } } }
